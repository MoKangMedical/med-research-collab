### 一、自动化脚本

#### 1. **数据下载自动化**
```python
import requests
import pandas as pd

def download_data(url, save_path):
    """下载数据并保存"""
    response = requests.get(url)
    with open(save_path, 'wb') as f:
        f.write(response.content)
    print(f"数据已保存到: {save_path}")

# 使用示例
download_data('https://example.com/data.csv', 'data/raw.csv')
```

#### 2. **分析流程自动化**
```python
import subprocess
import sys

def run_analysis_pipeline():
    """运行完整分析流程"""
    steps = [
        'python scripts/01_data_cleaning.py',
        'python scripts/02_data_analysis.py',
        'python scripts/03_visualization.py',
        'python scripts/04_report_generation.py'
    ]
    
    for step in steps:
        print(f"运行: {step}")
        result = subprocess.run(step, shell=True)
        if result.returncode != 0:
            print(f"错误: {step} 失败")
            sys.exit(1)
    
    print("✅ 分析流程完成")
```

def calculate_mean(data):
    """
    计算数据的平均值
    
    参数:
    data: list或numpy数组
        输入数据
    
    返回:
    float
        数据的平均值
    
    示例:
    >>> calculate_mean([1, 2, 3, 4, 5])
    3.0
    """
